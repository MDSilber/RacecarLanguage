make move around track
{

	:) go to the edge of the board (left side) 
	
	if can move left
	{
		turn left
	}
	repeat if can move forward = true
	{
		drive forward 1
	}
	set original_position = get car position
	
	:) start to go around the board
	
	if can move left = true
	{
		turn left
		move south original_position
	}
	else if move right = true
	{
		turn right
		move north original_position
	}
	else
	{
		drive backward
		move east original_position
	}
}
