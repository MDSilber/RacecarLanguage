define turnNinetyDegrees using dir (word)
{
    if dir is "left"
    {
        turn left
        turn left
    }
    elseIf dir is "right"
    {
        turn right
        turn right
    }
}

turn right

drive forward 10
 
turnNinetyDegrees "left"

turnNinetyDegrees "right"
print "going southeast again... "
drive forward 10

turnNinetyDegrees "right"
print "now heading southwest again... "
drive forward 10

turnNinetyDegrees "right"
